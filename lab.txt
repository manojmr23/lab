import 'package:flutter/material.dart';

void main() {
  runApp(const CGPACalculatorApp());
}

class CGPACalculatorApp extends StatelessWidget {
  const CGPACalculatorApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: "CGPA Calculator",
      theme: ThemeData(primarySwatch: Colors.blue),
      home: CGPACalculator(),
    );
  }
}

class CGPACalculator extends StatefulWidget {
  @override
  _CGPACalculatorState createState() => _CGPACalculatorState();
}

class _CGPACalculatorState extends State<CGPACalculator> {
  int subjectCount = 0;
  List<TextEditingController> gradeControllers = [];

  double? resultCGPA;

  void generateFields() {
    gradeControllers =
        List.generate(subjectCount, (index) => TextEditingController());
    setState(() {});
  }

  void calculateCGPA() {
    double total = 0;

    for (var controller in gradeControllers) {
      if (controller.text.isEmpty) {
        resultCGPA = null;
        setState(() {});
        return;
      }
      total += double.tryParse(controller.text) ?? 0;
    }

    resultCGPA = total / subjectCount;
    setState(() {});
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text("CGPA Calculator"),
        backgroundColor: Colors.blue,
      ),
      body: SingleChildScrollView(
        padding: const EdgeInsets.all(20),
        child: Column(
          children: [
            const Text(
              "Enter Number of Subjects",
              style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
            ),
            TextField(
              keyboardType: TextInputType.number,
              decoration: const InputDecoration(
                hintText: "e.g. 5",
                border: OutlineInputBorder(),
              ),
              onChanged: (value) {
                subjectCount = int.tryParse(value) ?? 0;
              },
            ),
            const SizedBox(height: 10),

            ElevatedButton(
              onPressed: () {
                generateFields();
              },
              child: const Text("Generate Fields"),
            ),

            const SizedBox(height: 20),

            // Dynamic Grade Input Fields
            Column(
              children: List.generate(
                gradeControllers.length,
                (index) => Padding(
                  padding: const EdgeInsets.only(bottom: 12),
                  child: TextField(
                    controller: gradeControllers[index],
                    keyboardType: TextInputType.number,
                    decoration: InputDecoration(
                      labelText: "Enter Grade Point for Subject ${index + 1}",
                      border: const OutlineInputBorder(),
                    ),
                  ),
                ),
              ),
            ),

            const SizedBox(height: 20),

            ElevatedButton(
              onPressed: calculateCGPA,
              child: const Text("Calculate CGPA"),
            ),

            const SizedBox(height: 20),

            // Result
            if (resultCGPA != null)
              Text(
                "Your CGPA is: ${resultCGPA!.toStringAsFixed(2)}",
                style: const TextStyle(
                    fontSize: 22,
                    fontWeight: FontWeight.bold,
                    color: Colors.green),
              ),
          ],
        ),
      ),
    );
  }
}
